1. Konsep & Justifikasi Monad
Game dalam 1 Kalimat
N pemain stake MON → countdown random 1-5 detik → "GO!" muncul → pemain dengan reaction time
tercepat menang seluruh pot.
Kenapa Ide Ini Lulus Filter "Mati di Ethereum, Hidup di Monad"
Reaction time matter dalam millidetik.  Kalau finality 12 detik (Ethereum), tap pemenang baru
confirmed setelah game lain selesai. Demo flat.
Monad finality 800ms  bikin gameplay terasa instant. Pemenang langsung ketahuan.
Aligned dengan thesis Monad: "skill-based risk-to-earn interfaces where players can wager on their own
outcomes — at an atomic and micro level, allowing thousands of competitions to be run."
Kenapa Demoable di Panggung
2-10 audience volunteer per match, 30 detik per match, 5 match = ramai dan kompetitif
Visual sederhana, paham dalam 5 detik
Pemenang dapat MON real, bisa dilihat di wallet mereka langsung
Bisa dijalankan multi-round = audience engagement total

pragma solidity ^0.8.20;
contract Reflex {
    enum MatchState { Open, Locked, Active, Finished }
    struct Match {
        address host;
        uint256 stakePerPlayer;
        uint8 maxPlayers;
        address[] players;
        uint256 goTimestampMs;
        uint256 settleDeadlineMs;
        address winner;
        uint256 winnerReactionMs;
        uint8 tappedCount;
        MatchState state;
    }
    mapping(uint256 => Match) public matches;
    mapping(uint256 => mapping(address => bool)) public hasTapped;
    mapping(address => uint256) public bestReactionMs;
    uint256 public matchCounter;
    uint8 public constant MIN_PLAYERS = 2;
    uint8 public constant MAX_PLAYERS_CAP = 2 0;
    event MatchCreated(uint256 indexed matchId, address host, uint256 stake, uint8 maxPlayers);
    event MatchJoined(uint256 indexed matchId, address player, uint8 currentCount);
    event MatchStarted(uint256 indexed matchId, uint256 goTimestampMs, uint8 totalPlayers);
    event TapSubmitted(uint256 indexed matchId, address player, uint256 reactionMs);
    event MatchFinished(uint256 indexed matchId, address winner, uint256 reactionMs, uint256 pot
    function createMatch(uint8 maxPlayers) external payable returns (uint256) {
        require(msg.value > 0, "Stake required");
        require(maxPlayers >= MIN_PLAYERS && maxPlayers <= MAX_PLAYERS_CAP, "Invalid count");
        matchCounter++;
        Match storage m = matches[matchCounter];
        m.host = msg.sender;
        m.stakePerPlayer = msg.value;
        m.maxPlayers = maxPlayers;
        m.state = MatchState.Open;
        m.players.push(msg.sender);
        emit MatchCreated(matchCounter, msg.sender, msg.value, maxPlayers);
        emit MatchJoined(matchCounter, msg.sender, 1);
        return matchCounter;
    }
    function joinMatch(uint256 matchId) external payable {
        Match storage m = matches[matchId];
        require(m.state == MatchState.Open, "Not joinable");
        require(msg.value == m.stakePerPlayer, "Stake mismatch");
        require(m.players.length < m.maxPlayers, "Match full");

        m.players.push(msg.sender);
        emit MatchJoined(matchId, msg.sender, uint8(m.players.length));
        if (m.players.length == m.maxPlayers) {
            m.state = MatchState.Locked;
        }

    function lockMatch(uint256 matchId) external {
        Match storage m = matches[matchId];
        require(msg.sender == m.host, "Only host");
        require(m.state == MatchState.Open, "Not open");
        require(m.players.length >= MIN_PLAYERS, "Need 2+ players");
        m.state = MatchState.Locked;
    }
    function startMatch(uint256 matchId) external {
        Match storage m = matches[matchId];
        require(msg.sender == m.host, "Only host");
        require(m.state == MatchState.Locked, "Not locked");
        m.goTimestampMs = block.timestamp * 1 0 0 0;
        m.settleDeadlineMs = m.goTimestampMs + 5 0 0 0;
        m.state = MatchState.Active;
        emit MatchStarted(matchId, m.goTimestampMs, uint8(m.players.length));
    }
    function submitTap(uint256 matchId, uint256 clientTimestampMs) external {
        Match storage m = matches[matchId];
        require(m.state == MatchState.Active, "Not active");
        require(!hasTapped[matchId][msg.sender], "Already tapped");
        require(_isPlayer(m, msg.sender), "Not a player");
        require(clientTimestampMs > m.goTimestampMs, "Too early");
        require(clientTimestampMs <= m.settleDeadlineMs, "Too late");
        uint256 reactionMs = clientTimestampMs - m.goTimestampMs;
        hasTapped[matchId][msg.sender] = true;
        m.tappedCount++;
        if (m.winner == address( 0) || reactionMs < m.winnerReactionMs) {
            m.winner = msg.sender;
            m.winnerReactionMs = reactionMs;
        }
        if (bestReactionMs[msg.sender] == 0 || reactionMs < bestReactionMs[msg.sender]) {
            bestReactionMs[msg.sender] = reactionMs;
        }
        emit TapSubmitted(matchId, msg.sender, reactionMs);
        if (m.tappedCount == m.players.length) {
            _settle(matchId);

    }
    function settleMatch(uint256 matchId) external {
        Match storage m = matches[matchId];
        require(m.state == MatchState.Active, "Not active");
        require(block.timestamp * 1 0 0 0 > m.settleDeadlineMs, "Too early");
        require(m.winner != address( 0), "No tap submitted");
        _settle(matchId);
    }
    function _settle(uint256 matchId) internal {
        Match storage m = matches[matchId];
        m.state = MatchState.Finished;
        uint256 pot = m.stakePerPlayer * m.players.length;
        (bool sent, ) = m.winner.call{value: pot}("");
        require(sent, "Payout failed");
        emit MatchFinished(matchId, m.winner, m.winnerReactionMs, pot);
    }
    function _isPlayer(Match storage m, address addr) internal view returns (bool) {
        for (uint i = 0; i < m.players.length; i++) {
            if (m.players[i] == addr) return true;
        }
        return false;
    }
    function getMatchPlayers(uint256 matchId) external view returns (address[] memory) {
        return matches[matchId].players;
    }
}

4. Game Flow Multi-player
7 / 11 Reflex MVP  — Monad Blitz5. Tantangan Sync N Device "GO!" Bersamaan
Ini bagian paling tricky di multi-player . Ada 3 pendekatan:
Pendekatan A — Pure on-chain
Frontend tunggu event MatchStarted  → tampilkan "GO!". Karena Monad finality 800ms, semua device
dapat event dalam 800ms range. Beda 200-300ms antar device acceptable karena measurement reaction
time mulai dari goTimestampMs  yang sama.
Pendekatan B — Hybrid RECOMMENDED
Host trigger startMatch() . Frontend semua pemain listen event. Saat event masuk, hitung reaction time
dari event.goTimestampMs  (bukan Date.now()  saat dilihat di layar). Fairness sempurna  karena baseline
waktu sama untuk semua.
Pendekatan C — Off-chain coordination
Pakai W ebSocket/Pusher untuk broadcast "GO!". Lebih cepat tapi butuh server . Skip untuk 2 jam.Step 1 — Host membuat match
createMatch(maxPlayers=10)  + stake 0.01 MON → matchId = 5
Step 2 — Share link
Host share link /match/5  ke audience (via QR code di slide)
Step 3 — Players join
Audience tap link, panggil joinMatch(5)  + stake. Event MatchJoined  emit, semua frontend update
count "3/10"
Step 4 — Host start game
Host klik "Start Game" → lockMatch(5)  [optional kalau belum full] → frontend countdown random 1-5
detik visual ("W AIT...") → startMatch(5)
Step 5 — TAP race
Event MatchStarted  masuk → semua frontend tampilkan "T AP NOW!" hijau full-screen → players tap
→ submitTap(5, Date.now())
Step 6 — Auto-settle
Setelah semua tap atau timeout 5 detik → event MatchFinished  → pemenang dapat pot

Frontend listen MatchStarted  via watchContractEvent  dengan pollingInterval: 200
Saat event masuk, save goTimestampMs  di state
User tap → submitTap(matchId, Date.now())
Reaction time dihitung di contract: clientTimestampMs - goTimestampMs

9. Catatan Teknis Tambahan
Hal yang Sering Bikin Gagal di 2 Jam
Masalah Solusi
Wallet connection issue Pakai Privy lebih cepat dari WalletConnect; pakai 1 metode saja
RPC rate limit Pakai dedicated RPC (Alchemy free tier untuk Monad), jangan public RPC
Event listening lambat Set pollingInterval: 500  (atau 200) di config viem — default 4s terlalu lambat
Race condition tap Untuk N-player: contract handle "fastest wins", bukan "first tx wins" — tidak masalah
Storage gas players[]  push ~20k gas. OK untuk N≤20. Jangan 100+.
RPC Load untuk N Device
N device polling event setiap 200ms = N×5 req/sec. Untuk N=10, itu 50 req/sec. Pakai dedicated RPC
(Alchemy/Chainstack), free tier cukup.
